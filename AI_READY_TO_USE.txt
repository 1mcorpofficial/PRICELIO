// services/api/src/index.js
app.post('/baskets/:id/optimize-ai', async (req, res) => {
  const optimization = await fetch('http://localhost:3001/ai/optimize-basket', {
    method: 'POST',
    body: JSON.stringify(req.body)
  });
  res.json(await optimization.json());
});
